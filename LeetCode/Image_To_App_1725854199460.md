```html
<html>
<head>
    <script src="https://cdn.tailwindcss.com"></script>
    <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/5.15.3/css/all.min.css">
</head>
<body class="bg-gray-900 flex items-center justify-center min-h-screen">
    <div class="text-center">
        <h1 class="text-white text-3xl mb-8">Where knowledge begins</h1>
        <div class="bg-gray-800 p-4 rounded-lg flex items-center space-x-4 relative">
            <input id="searchInput" type="text" placeholder="Ask anything..." class="bg-transparent text-gray-400 flex-grow focus:outline-none">
            <div class="flex items-center space-x-2 cursor-pointer" id="focusButton">
                <i class="fas fa-stream text-gray-400"></i>
                <span id="focusText" class="text-gray-400">Focus</span>
            </div>
            <div class="flex items-center space-x-2 cursor-pointer" id="attachButton">
                <i class="fas fa-paperclip text-gray-400"></i>
                <span class="text-gray-400">Attach</span>
                <input type="file" id="fileInput" class="hidden" accept="image/*">
            </div>
            <div class="flex items-center space-x-2">
                <label class="switch">
                    <input id="proSwitch" type="checkbox" checked>
                    <span class="slider round"></span>
                </label>
                <span class="text-cyan-400">Pro</span>
            </div>
            <button id="searchButton" class="text-gray-400">
                <i class="fas fa-arrow-right"></i>
            </button>
            <div id="focusOptions" class="absolute top-full left-0 mt-2 w-full bg-gray-700 p-4 rounded-lg hidden">
                <div class="grid grid-cols-3 gap-4">
                    <div class="focus-option text-white cursor-pointer">Network</div>
                    <div class="focus-option text-white cursor-pointer">Academic</div>
                    <div class="focus-option text-white cursor-pointer">Math</div>
                    <div class="focus-option text-white cursor-pointer">Writing</div>
                    <div class="focus-option text-white cursor-pointer">Video</div>
                    <div class="focus-option text-white cursor-pointer">Social</div>
                </div>
            </div>
        </div>
        <div id="analyzingText" class="text-white text-xl mt-4 hidden">Analyzing image...</div>
        <div id="featureMessage" class="text-white text-xl mt-4 hidden">Features still being implemented, please choose another...</div>
        <div id="essayContent" class="text-white text-xl mt-4 hidden"></div>
    </div>

    <style>
        .switch {
            position: relative;
            display: inline-block;
            width: 34px;
            height: 20px;
        }

        .switch input {
            opacity: 0;
            width: 0;
            height: 0;
        }

        .slider {
            position: absolute;
            cursor: pointer;
            top: 0;
            left: 0;
            right: 0;
            bottom: 0;
            background-color: #ccc;
            transition: .4s;
            border-radius: 34px;
        }

        .slider:before {
            position: absolute;
            content: "";
            height: 12px;
            width: 12px;
            left: 4px;
            bottom: 4px;
            background-color: white;
            transition: .4s;
            border-radius: 50%;
        }

        input:checked + .slider {
            background-color: #06b6d4;
        }

        input:checked + .slider:before {
            transform: translateX(14px);
        }
    </style>

    <script>
        let currentFocus = 'Network';

        document.getElementById('searchButton').addEventListener('click', function() {
            performSearch();
        });

        document.getElementById('searchInput').addEventListener('keypress', function(event) {
            if (event.key === 'Enter') {
                performSearch();
            }
        });

        document.getElementById('attachButton').addEventListener('click', function() {
            document.getElementById('fileInput').click();
        });

        document.getElementById('fileInput').addEventListener('change', function(event) {
            if (event.target.files.length > 0) {
                document.getElementById('analyzingText').classList.remove('hidden');
                const file = event.target.files[0];
                const reader = new FileReader();
                reader.onload = function(e) {
                    const imageUrl = e.target.result;
                    analyzeImage(imageUrl);
                };
                reader.readAsDataURL(file);
            }
        });

        document.getElementById('focusButton').addEventListener('click', function() {
            const focusOptions = document.getElementById('focusOptions');
            focusOptions.classList.toggle('hidden');
        });

        document.querySelectorAll('.focus-option').forEach(option => {
            option.addEventListener('click', function() {
                const focusText = document.getElementById('focusText');
                focusText.textContent = this.textContent;
                currentFocus = this.textContent;
                document.getElementById('focusOptions').classList.add('hidden');
                updatePlaceholder();
            });
        });

        function updatePlaceholder() {
            const searchInput = document.getElementById('searchInput');
            if (currentFocus === 'Math') {
                searchInput.placeholder = 'Enter your mathematical calculation...';
            } else if (currentFocus === 'Writing') {
                searchInput.placeholder = 'Write the topic of the essay...';
            } else {
                searchInput.placeholder = 'Ask anything...';
            }
        }

        function performSearch() {
            const query = document.getElementById('searchInput').value;
            const isPro = document.getElementById('proSwitch').checked;
            let searchUrl;

            switch (currentFocus) {
                case 'Network':
                    searchUrl = isPro ? `https://www.google.com/search?q=${query}` : `https://www.bing.com/search?q=${query}`;
                    break;
                case 'Video':
                    searchUrl = `https://www.youtube.com/results?search_query=${query}`;
                    break;
                case 'Academic':
                    searchUrl = `https://scholar.google.com/scholar?q=${query}`;
                    break;
                case 'Social':
                    searchUrl = `https://www.reddit.com/search/?q=${query}`;
                    break;
                case 'Math':
                    solveMath(query);
                    return;
                case 'Writing':
                    generateEssay(query);
                    return;
                default:
                    searchUrl = isPro ? `https://www.google.com/search?q=${query}` : `https://www.bing.com/search?q=${query}`;
            }

            window.location.href = searchUrl;
        }

        function solveMath(query) {
            const apiUrl = `https://api.mathjs.org/v4/?expr=${encodeURIComponent(query)}`;
            fetch(apiUrl)
                .then(response => response.text())
                .then(result => {
                    if (result.includes('Error')) {
                        const wolframAlphaUrl = `https://www.wolframalpha.com/input/?i=${encodeURIComponent(query)}`;
                        window.location.href = wolframAlphaUrl;
                    } else {
                        alert(`Result: ${result}`);
                    }
                })
                .catch(error => {
                    const wolframAlphaUrl = `https://www.wolframalpha.com/input/?i=${encodeURIComponent(query)}`;
                    window.location.href = wolframAlphaUrl;
                });
        }

        function generateEssay(topic) {
            const essayContent = document.getElementById('essayContent');
            essayContent.classList.remove('hidden');
            essayContent.textContent = `Essay on "${topic}" is being generated...`;

            const apiUrl = `https://api.example.com/generate-essay?topic=${encodeURIComponent(topic)}`;
            fetch(apiUrl)
                .then(response => response.json())
                .then(data => {
                    essayContent.textContent = data.essay;
                })
                .catch(error => {
                    essayContent.textContent = 'Error generating the essay. Please try again later.';
                });
        }

        function analyzeImage(imageUrl) {
            const tinEyeUrl = `https://tineye.com/search?url=${encodeURIComponent(imageUrl)}`;
            window.location.href = tinEyeUrl;
        }
    </script>
</body>
</html>
```
